 24class GenerateSinglePopulationWrapper:

 44class GenerateBitstreamPopulation:
 45    # force everything to be passed by keyword
 46    def __init__(self, *, sz: int, bitstream_sz: int,
 47                 post_construction_strategy: PostConstructionStrategy,
 48                 randomization_strategy: RandomizationStrategy,
 49                 mutation_prob: float, rand: Random):
 50        self.__sz = sz
 51        self.__bitstream_sz = bitstream_sz
 52        self.__post_construction_strategy = post_construction_strategy
 53        self.__randomization_strategy = randomization_strategy
 54        self.__rand = rand
 55
 56        self.__mutation_prob = mutation_prob
 57        
 58    def generate(self) -> Population:
 59        """
 60        Creates initial population based on the config.
 61        1. Clears the files used to keep track of circuit
 62        2. Uses appropriate initialization method specified by config.
 63        3. Handles randomization until condition in config is met.
 64        """
 65
 66        individuals = []
 67
 68        for index in range(1, self.__sz + 1):
 69            individual = BitstreamIndividual(self.__bitstream_sz, self.__rand, self.__mutation_prob)
 70            individual = self.__post_construction_strategy.run(individual)
 71
 72            individuals.append(individual)
 73
 74        circuits = self.__randomization_strategy.randomize(individuals)
 75
 76        return Population(circuits, None)

 99class FileBasedCircuitFactory:
100    def __init__(self, *, sz: int, logger: Logger, directories: Directories, routing_type: str, accessed_columns: list[int]):
101        self.__logger = logger
102        self.__directories = directories
103        self.__routing_type = routing_type
104        self.__accessed_columns = accessed_columns
105
106        wipe_folder(self.__directories.asc_dir)
107        wipe_folder(self.__directories.bin_dir)
108        wipe_folder(self.__directories.data_dir)
109
110        self.__circuits = []
111        for i in range(sz):
112            circuit = FileBasedCircuit(
113                index=i,
114                filename="hardware" + str(i),
115                template=SEED_HARDWARE_FILEPATH,
116                logger=self.__logger,
117                directories=self.__directories,
118                routing_type=self.__routing_type,
119                accessed_columns=self.__accessed_columns
120            )
121            self.__circuits.append(circuit)
122
123    def create(self, populations: list[Population]) -> dict[Circuit, list[tuple[Population, Individual]]]:
124        res: dict[Circuit, list[tuple[Population, Individual]]] = {}
125        index = 0
126        for p in populations:
127            pop = p.population_list
128            for (individual, _) in pop:
129                circuit = self.__circuits[index]
130                bitstream: list[bool] = individual.get_bitstream() # type: ignore
131                circuit.set_bitstream(bitstream)
132                res[circuit] = [(p, individual)]
133                index += 1
134
135        return res
