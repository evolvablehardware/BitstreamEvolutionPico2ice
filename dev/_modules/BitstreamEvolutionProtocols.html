 44class GenDataIncrementer():
 45    def __init__(self, max_gen_num:int):
 46        self.__max_gen_num = max_gen_num

110class Population:
111    """
112    This is the Population object used to hold individuals and their fitnesses durring evolution. 
113    It starts out with its full list of individuals, and optionally fitnesses.
114    No individuals in the list may be duplicates. (determined using == )
115    If there is no fitness it is None. 
116    Fitnesses can be added to individuals in the population as desired, and once all individuals are added, the population can be sorted. 
117    The Population can also be itterated through. (iter() -then-> next())
118    """
119
120    # May want specific type variables.

121    def __init__(self,individuals: Iterable[Individual], fitnesses:Optional[Iterable[Fitness|None]]=None):
122        "Can raise ValueError if Individuals are not unique."
123        ind = list(individuals)
124
125        if len(set(ind)) != len(ind):
126            raise ValueError("There are duplicate individuals in this population. Each Individual in a population must be a unique object.")
127
128        fit_list = list(fitnesses) if fitnesses is not None else [None]*len(ind)
129        fit = fit_list if len(fit_list) == len(ind) else [None]*len(ind)
130
131        self.population_list:list[tuple[Individual,Optional[Fitness]]] = list(zip(ind,fit))


132        # = [(Individual, Fitness), (Individual2, Fitness2), ...]
133
134    def __iter__(self)->Iterator[tuple[Individual,Optional[Fitness]]]:
135        # call iter() to get iterator, then next()
136        # If wanted to be safe, return a copy that can't change
137        return iter(self.population_list)
138    
139    def __len__(self)->int:
140        return len(self.population_list)

145    def set_fitness(self, individual:Individual, fitness:Fitness)->None:
146        "This can return value error if provided individual is not in the population."
147        for i,if_tup in enumerate(self.population_list):
148            if if_tup[0] == individual:
149                self.set_fitness_by_index(index=i,fitness=fitness)
150                return
151        raise ValueError(f"Cound not find provided Individual in the population. Individual: {individual}")

158    def sort(self,key:Callable[[Fitness],Fitness],reverse:bool)->None:
159        "Operates on the Population fitness function like the standard sort for a list. May Raise TypeError if population not fully evaluated."
160        try:
161            self.population_list.sort(
162                key=lambda if_tup: key(if_tup[1]), # type: ignore
163                reverse= reverse
164            )
165        except TypeError:
166            raise TypeError(f"Population does not have all individual's fitnesses fully specified. {sum([t[1] is None for t in self.population_list])} individuals did not have a fitness assigned.")
