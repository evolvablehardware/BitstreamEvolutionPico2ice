33    def run(self):
34        '''
35        Runs the desired experiment, based on the protocols provided
36        '''
37        populations: list[Population] = self.__gen_init_populations()
38        gen_data = self.__gen_data_factory(None)
39        while gen_data is not None:
40            measurements = self.__generate_measurements(self.__circuit_factory, populations)
41            tasks = [self.__hardware.request_measurement(m) for m in measurements]
42            results = asyncio.run( asyncio.gather(*tasks) ) # I added asyncio.run to make sure that the async experiements were run at this point
43            # Maybe figure out task groups. See https://docs.python.org/3/library/asyncio-task.html#coroutines-and-tasks
44
45            for p in populations:
46                self.__eval_population_fitness(p, results)
47            populations = list(map(lambda p: self.__reproduce(p), populations))
48
49            fits: list[float] = []
50            for p in populations:
51                for (i, f) in p:
52                    fits.append(f) # type: ignore
53            # TODO: calculate diversity
54            self.__plot_data_recorder.record_generation(fits, gen_data.generation_number, 0)
55
56            gen_data = self.__gen_data_factory(gen_data)
