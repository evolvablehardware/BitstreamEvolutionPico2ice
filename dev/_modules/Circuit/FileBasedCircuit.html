 88    def __run_at_each_modifiable(self, lambda_func, hardware_file = None, accessible_columns = None,
 89        routing_type=None):
 90        """
 91        Runs the lambda_func at every modifiable position
 92        Args passed to lambda_func: value of bit (as a byte), row, col
 93        If lambda_func returns a value (byte), then the bit is set to that number
 94        If lambda_func returns None, then the bit is left unmodified
 95        Keep in mind the bytes are the ASCII codes, so for example 49 = 1
 96
 97        .. warning::
 98            Go over this with someone who can clarify what all of the data types are. 
 99
100        Parameters
101        ----------
102        lambda_func : Callable
103            This function is called for each modifiable bit. The bit value is passed into lambda_func.
104            If lambda_func returns None, the bit is left unmodified. If lambda_func returns another value,
105            then the bit at that position is replaced with the return value of lambda_func.
106        hardware_file : str | None
107            The path to the hardware file to read from/write to. If no value provided, uses this Circuit's hardware file.
108        accessible_columns : list[str] | None
109            The accessible columns. If no value provided, uses the current configuration value.
110        routing_type: str | None
111            The routing type (MOORE or NEWSE). If no value provided, uses the current configuration value.
112        """
113
114        if hardware_file is None:
115            hardware_file = self._hardware_file
116
117        if accessible_columns is None:
118            accessible_columns = self.__accessed_columns
119
120        if routing_type is None:
121            routing_type = self.__routing_type
122
123        # Set tile to the first location of the substring ".logic_tile"
124        # The b prefix makes the string an instance of the "bytes" type
125        # The .logic_tile header indicates that there is a tile, so the "tile" variable stores the starting point of the current tile
126        tile = hardware_file.find(b".logic_tile")
127        
128        while tile > 0:
129            # Set pos to the position of this tile, but with the length of ".logic_tile" added so it is in front of where we have the x/y coords
130            pos = tile + len(".logic_tile")
131            
132            # Check if the position is legal to modify
133            if self.__tile_is_included(hardware_file, pos):
134                # Find the start and end of the line; the positions of the \n newline just before and at the end of this line
135                # The start is the newline position + 1, so the first valid bit character
136                # line_size is self-explanatory
137                # This finds the length of a standard line of bits (so the width of each data-containing line in this tile)
138                line_start = hardware_file.find(b"\n", tile) + 1
139                line_end = hardware_file.find(b"\n", line_start + 1)
140                line_size = line_end - line_start + 1
141
142                # Determine which rows we can modify
143                # TODO ALIFE2021 The routing protocol here is dated and needs to mimic that of the Tone Discriminator
144                rows = []
145                if routing_type == "MOORE":
146                    rows = [1, 2, 13]
147                elif routing_type == "NEWSE":
148                    rows = [1, 2]
149                # Iterate over each row and the columns that we can access within each row
150                for row in rows:
151                    for col in accessible_columns:
152                        # This will get us to individual bits. If the mutation probability passes
153                        # Our position is now going to be the start of the first line, plus the line size multiplied to get to our desired row,
154                        # and finally added to the column (with the int cast to sanitize user input)
155                        pos = line_start + line_size * (row - 1) + int(col)
156                        bit_value = hardware_file[pos]
157                        lambda_return = lambda_func(bit_value, row, col)
158                        if lambda_return is not None:
159                            # need to re-assign the bit
160                            hardware_file[pos] = lambda_return
161
162            # Find the next logic tile, and start again
163            # Will return -1 if .logic_tile isn't found, and the while loop will exit
164            tile = hardware_file.find(b".logic_tile", tile + 1)

185    def __tile_is_included(self, hardware_file, pos):
186        """
187        Determines whether a given tile is available for modificiation.
188        NOTE: Tile = the .logic_tile in the asc file.
189
190        .. warning::
191            Preexisting todo: Replace magic values with a more generalized solution. 
192            These magic values are indicative of the underlying hardware (ice40kh1k)
193
194        Parameters
195        ----------
196        hardware_file : mmap
197            Memory Mapped hardware file
198        pos : int
199            Index of the first byte in the .asc file for the hardware
200
201        Returns
202        -------
203        bool
204            True if the tile at that position is valid (The Tiles we can modigy)
205        """
206        # Replace these magic values with a more generalized solution
207        # Magic values are indicative of the underlying hardware (ice40hx1k)
208        # A different model will require different magic values (i.e. ice40hx8k)
209        VALID_TILE_X = range(4, 10)
210        VALID_TILE_Y = range(1, 17)
211
212        # NOTE x and y are stored as ints to aid the loops that search and identify
213        # tiles while scraping the asc files
214        # This is in the actual asc file; this is why we can simply pull from "pos"
215        # i.e. you'll see the header ".logic_file 1 1" - x=1, y=1
216        
217        # This is where we had a fundamental issue before: The value in the hardware at this position is going to be an ASCII char value, not the actual
218        # number. Here, we parse the byte as an integer, then to a char, then back to an integer
219        
220        # However, we have a great problem now: what about multi-digit numbers?
221        # Find the space that separates the x and y, and find the end of the line
222        # Then, grab the bytes for x, grab the bytes for y, convert to strings, and parse those strings
223        space_pos = hardware_file.find(b" ", pos + 1)
224        eol_pos = hardware_file.find(b"\n", pos)
225        x_bytes = hardware_file[pos:space_pos]
226        y_bytes = hardware_file[space_pos:eol_pos]
227        x_str = x_bytes.decode("utf-8").strip()
228        y_str = y_bytes.decode("utf-8").strip()
229        x = int(x_str)
230        y = int(y_str)
231        is_x_valid = x in VALID_TILE_X
232        is_y_valid = y in VALID_TILE_Y
233
234        return is_x_valid and is_y_valid

237        bitstream: list[bool] = []

239            if bit == 48:

241            elif bit == 49:
242                bitstream.append(True)

259    def get_file_attribute_st(mmapped_file, attribute):
260        '''
261        Returns the value of the stored attribute from the hardware file.
262        Circuits are capable of storing string name-value pairs in their hardware file, for purposes such as
263        tracking most recently-evaluated fitness of a Circuit
264        Static version of get_file_attribute that requires the memory-mapped file to be provided
265
266        Parameters
267        ----------
268        mmapped_file : mmap
269            The memory-mapped hardware file of the circuit
270        attribute : str
271            Attribute name to lookup
272
273        Returns
274        -------
275        str
276            File attribute value
277        '''
278        index = mmapped_file.find(b".comment FILE_ATTRIBUTES")
279        if index < 0:
280            return '0'
281        else:
282            newline_index = mmapped_file.find(b'\n', index)
283            searchable_area = mmapped_file[index:newline_index]
284            attr_index = searchable_area.find(bytes(attribute + '={', 'utf-8'))
285            if attr_index < 0: # Value doesn't exist yet
286                return '0'
287            attr_index = attr_index + len(attribute + "={")
288            end_index = searchable_area.find(b'}', attr_index)
289            value_bytes = searchable_area[attr_index:end_index]
290            return str(value_bytes, 'utf-8')

293    def set_file_attribute_st(hardware_file, attribute, value):
294        '''
295        Sets a Circuit's file attribute to the specified value
296        Circuits are capable of storing string name-value pairs in their hardware file, for purposes such as
297        tracking most recently-evaluated fitness of a Circuit
298        Static version of set_file_attribute that requires the memory-mapped file to be provided
299
300        Parameters
301        ----------
302        hardware_file : mmap
303            The memory-mapped hardware file of the Circuit
304        attribute : str
305            The name of the attribute to modify
306        value : str
307            The value to assign to the attribute
308        '''
309        # Check if the comment exists
310        #hardware_file = open(file_path, "r+")
311        mmapped_file = mmap(hardware_file.fileno(), 0)
312        index = mmapped_file.find(b".comment FILE_ATTRIBUTES")
313        if index < 0:
314            # Create the comment
315            comment_line = ".comment FILE_ATTRIBUTES " + attribute + "={" + value + "}\n"
316            # This requires re-mapping the self._hardware_file
317            #hardware_file = open(file_path, "r+")
318            content = hardware_file.read()
319            hardware_file.seek(0, 0)
320            hardware_file.write(comment_line + content)
321        else:
322            # Check if the attribute exists
323            end_index = mmapped_file.find(b'\n', index)
324            line = str(mmapped_file[index:end_index], 'utf-8')
325            attr_index = line.find(attribute + "={")
326            lines = hardware_file.readlines()
327            line_index = 0 # Index of the line that contains the attribute comment
328            for l in lines:
329                if l.find(".comment FILE_ATTRIBUTES") >= 0:
330                    break
331                line_index = line_index + 1
332
333            if attr_index < 0:
334                # Attribute doesn't exist yet
335                line = line + " " + attribute + "={" + value + "}\n"
336            else:
337                attr_end_index = line.find('}', attr_index) + 2
338                before_attr = line[:attr_index]
339                after_attr = line[attr_end_index:]
340                line = before_attr + attribute + "={" + value + "} " + after_attr + '\n'
341            lines[line_index] = line
342            hardware_file.truncate(0)
343            hardware_file.seek(0)
344            hardware_file.writelines(lines)
