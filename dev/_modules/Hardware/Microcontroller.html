 16        self.__fpga = fpga

 27    async def request_measurement(self, measurement: Measurement) -> Measurement:
 28        ckt: Circuit = measurement.circuit
 29        fpga_data = FPGA_Compilation_Data(FPGA_Model.ICE40, self.__fpga)
 30        if measurement.data_request == DataRequest.WAVEFORM:
 31            ckt.compile(fpga_data)
 32            try:
 33                waveform = await self.measure_signal()
 34                measurement.result = Ok(waveform)
 35            except Exception as e:
 36                measurement.result = Err(e)
 37        elif measurement.data_request == DataRequest.OSCILLATIONS:
 38            ckt.compile(fpga_data)
 39            try:
 40                data = await self.measure_pulses(measurement.num_samples)
 41                measurement.result = Ok(data)
 42            except Exception as e:
 43                measurement.result = Err(e)
 44        # TODO: elif cases for other measurement types...
 45        return measurement

 50    async def measure_signal(self) -> list[int]:
 51        buf = []
 52
 53        self.__serial.reset_input_buffer()
 54        self.__serial.reset_output_buffer()
 55        self.__logger.log_event(1, "Reading microcontroller.")
 56        # The MCU is expecting a string '2' to initiate the ADC capture from the FPGA (waveform as opposed to pulses)
 57        self.__serial.write(b'2')
 58        line = self.__serial.read()
 59
 60        start = time()
 61
 62        # The MCU returns a START line followed by many lines of data (500 currently) followed by a FINISHED line
 63        while b"START\n" not in line:
 64            self.__serial.write(b'2')
 65            line = self.__serial.read_until()
 66
 67            if (time() - start) >= self.__config.read_timeout:
 68                self.__logger.log_warning(1, "Did not read START from MCU")
 69                self.__logger.log_warning(1, "Time Exceeded. Halting MCU Reading.")
 70                break
 71
 72        # TODO  This whole section can probably be optimized
 73        # Reads in samples from MCU, with each being 10microseconds apart
 74        while (b"FINISHED\n" not in line):
 75            line = self.__serial.read_until()
 76            if line != b"\n" and line != b"START\n" and line != b"FINISHED\n" and line != b"FINISHED\n":
 77                buf.append(line)
 78            if (time() - start) >= self.__config.read_timeout:
 79                self.__logger.log_warning(1, "Time Exceeded. Halting MCU Reading.")
 80                break
 81
 82        self.__logger.log_event(2, "Finished reading microcontroller. Logging data to file.")
 83
 84        waveform: list[int] = []
 85        for i in buf:
 86            if b"FINISHED" not in i:
 87                waveform.append(int(i))
 88
 89        self.__logger.log_event(2, "Completed reading waveform")
 90        return waveform

 92    async def measure_pulses(self, samples: int) -> list[int]:
 93        result = []
 94        for i in range(samples):
 95            result.extend(await self.measure_pulses_once())
 96        return result

 98    async def measure_pulses_once(self) -> list[int]:
 99        buf = []
100        # Poll serial line until START signal
101        self.__logger.log_event(3, f"Starting loop for reading")
102        
103        self.__serial.reset_input_buffer()
104        self.__serial.reset_output_buffer()
105        # NOTE The MCU is expecting a string '1' if fitness isn't measured this may be why
106        self.__serial.write(b'1') 
107        start = time()
108        self.__logger.log_event(3, f"Starting MCU loop...")
109
110        max_attempts = 5
111        attempts = 0
112        while True:
113            attempts = attempts + 1
114            self.__logger.log_event(3, f"Serial reading...")
115            p = self.__serial.read_until()
116            self.__logger.log_event(3, f"Serial read done")
117            if (time() - start) >= self.__config.read_timeout:
118                self.__logger.log_warning(1, f"Time Exceeded")
119                if attempts >= max_attempts:
120                    self.__logger.log_warning(3, f"Exceeded max attempts ({max_attempts}). Halting MCU reading")
121                    buf.append(-1)
122                    break
123            # TODO We should be able to do whatever this line does better
124            # This is currently doing a poor job at REGEXing the MCU serial return - can be done better
125            # It's supposed to handle exceptions from transmission loss (i.e. dropped or additional spaces, shifted colons, etc)
126            self.__logger.log_event(3, "Pulled", p, f"from MCU")
127            if (p != b"" and b":" not in p and b"START" not in p and b"FINISH" not in p and b" " not in p):
128                p = p.translate(None, b"\r\n")
129                buf.append(p)
130                break
131
132        end = time() - start
133
134        self.__logger.log_event(2, 'Length of buffer:', len(buf))
135        if len(buf) == 0:
136            buf.append(-1000) # This should never happen
137
138        result = []
139        for i in range(len(buf)):
140            self.__logger.log_event(2, f'Buffer entry {i}:', buf[i])
141            try:
142                result.append(int(buf[i]))
143            except ValueError:
144                result.append(-2)
145        
146        return result
