166    def log_monitor(self, prefix,  *msg):
167        if self.__config.save_log:
168            now = datetime.now()
169            print(now, prefix, *msg, file=self.__monitor_file)
170
171    def log_event(self, level, *msg):
172        if self.__config.log_level >= level:
173            print(*msg, file=self.__log_file)
174            self.log_monitor("", *msg)
175
176    def log_info(self, level, *msg):
177        if self.__config.log_level >= level:
178            print("INFO: ", OKBLUE, *msg, ENDC, file=self.__log_file)
179            self.log_monitor("INFO: ", *msg)
180
181    def log_warning(self, level, *msg):
182        if self.__config.log_level >= level:
183            print("WARNING: ", WARNING, *msg, ENDC, file=self.__log_file)
184            self.log_monitor("WARNING: ", *msg)
185
186    def log_error(self, level, *msg):
187        if self.__config.log_level >= level:
188            print("ERROR: ", FAIL, *msg, ENDC, file=self.__log_file)
189            self.log_monitor("ERROR: ", *msg)
190
191    def log_critical(self, level, *msg):
192        if self.__config.log_level >= level:
193            print("CRITICAL: ", FAIL, *msg, ENDC, file=self.__log_file)
194            self.log_monitor("CRITICAL: ", *msg)
